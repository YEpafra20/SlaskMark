% MATLAB Project: Data Visualization (Line Plot, Histogram, Scatter Plot)

% Clear workspace and command window
clear; clc; close all;

% Generate Sample Data
x = linspace(0, 10, 100); % X-axis values
y = sin(x); % Y values for sine wave (line plot)
data = randn(1, 1000); % Random data for histogram
x_scatter = rand(1, 50) * 10; % Random x values for scatter plot
y_scatter = rand(1, 50) * 10; % Random y values for scatter plot

% Create a Figure Window
figure;

% Line Plot
subplot(3,1,1); % Create first subplot (row, column, index)
plot(x, y, 'b', 'LineWidth', 2); % Plot sine wave
grid on;
title('Line Plot - Sine Wave');
xlabel('X values');
ylabel('Y values');

% Histogram
subplot(3,1,2); % Create second subplot
histogram(data, 30, 'FaceColor', 'r'); % Create histogram with 30 bins
grid on;
title('Histogram - Random Data Distribution');
xlabel('Data Values');
ylabel('Frequency');

% Scatter Plot
subplot(3,1,3); % Create third subplot
scatter(x_scatter, y_scatter, 50, 'filled', 'MarkerFaceColor', 'g'); % Create scatter plot
grid on;
title('Scatter Plot - Random Points');
xlabel('X values');
ylabel('Y values');

% Display the Figure
sgtitle('MATLAB Data Visualization Examples');
